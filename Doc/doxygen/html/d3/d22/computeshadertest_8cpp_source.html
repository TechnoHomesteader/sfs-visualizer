36  GLuint genComputeProg(GLuint texHandle) {
37  // Creating the compute shader, and the program object containing the shader
38  GLuint progHandle = glCreateProgram();
39  GLuint cs = glCreateShader(GL_COMPUTE_SHADER);
40 
41  // In order to write to a texture, we have to introduce it as image2D.
42  // local_size_x/y/z layout variables define the work group size.
43  // gl_GlobalInvocationID is a uvec3 variable giving the global ID of the thread,
44  // gl_LocalInvocationID is the local index within the work group, and
45  // gl_WorkGroupID is the work group's index
46 
47  const char *csSrc[] = {
48  "#version 430\n",
49  "uniform float roll;\
50  uniform image3D destTex;\
51  layout (local_size_x =4 , local_size_y = 4,local_size_z = 4) in;\
52  void main() {\
53  ivec3 storePos = ivec3(gl_GlobalInvocationID.xyz);\
54  float localCoef = length(vec2(ivec2(gl_LocalInvocationID.xyz)-8)/8.0);\
55  float globalCoef = sin(float(gl_WorkGroupID.x+gl_WorkGroupID.y+gl_WorkGroupID.z)*0.1 + roll)*0.5;\
56  imageStore(destTex, storePos, vec4(sin(roll*0.5+storePos.y),1.0, cos(roll*0.5+storePos.x),1.0));\
57  }"
58  };
59 
60 
61  glShaderSource(cs, 2, csSrc, NULL);
62  glCompileShader(cs);
63  int rvalue;
64  glGetShaderiv(cs, GL_COMPILE_STATUS, &rvalue);
65  if (!rvalue) {
66  fprintf(stderr, "Error in compiling the compute shader\n");
67  GLchar log[10240];
68  GLsizei length;
69  glGetShaderInfoLog(cs, 10239, &length, log);
70  fprintf(stderr, "Compiler log:\n%s\n", log);
71  exit(40);
72  }
73  glAttachShader(progHandle, cs);
74 
75  glLinkProgram(progHandle);
76  glGetProgramiv(progHandle, GL_LINK_STATUS, &rvalue);
77  if (!rvalue) {
78  fprintf(stderr, "Error in linking compute shader program\n");
79  GLchar log[10240];
80  GLsizei length;
81  glGetProgramInfoLog(progHandle, 10239, &length, log);
82  fprintf(stderr, "Linker log:\n%s\n", log);
83  exit(41);
84  }
85  glUseProgram(progHandle);
86 
87  glUniform1i(glGetUniformLocation(progHandle, "destTex"), 0);
88 
89  checkErrors("Compute shader");
90  return progHandle;
91  }
