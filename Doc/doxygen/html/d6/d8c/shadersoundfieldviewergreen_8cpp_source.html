135  void ShaderSoundfieldViewerGreen::loadSourcesFromCSV()
136  {
137  if (Application::instance().getCommandlineOptions().count("sourcedefinition") == 0)

139  std::cout << "no sourcedefinitions found-> using (default) sources on circle" <<std::endl;

142 
143  std::string strFileName = Application::instance().getCommandlineOptions()["sourcedefinition"].as<std::string >();
144  std::cout << "using Sourcedefitions from '"<< strFileName << "'" <<std::endl;
145  std::string line;
146  std::ifstream myfile (strFileName);
147  std::stringstream outstream;
148  std::vector<float> sources;
149  char delimiter = ',';
150  if (!myfile.is_open())
151  throw std::bad_exception("Sourcedefitions not found");
152 
153  int lineCount =0;
154  while ( getline (myfile,line) )
155  {
156  //read the entries
157  std::stringstream ss(line);
158  std::string item;
159  std::vector<std::string> elems;
160  while (std::getline(ss, item, delimiter)) {
161  elems.push_back(item);
162  }
163  if (elems[0]!="x" && elems.size() == 7) //ignore header line and empty lines
164  {
165  sources.push_back((float) atof(elems[0].c_str()));
166  sources.push_back((float) atof(elems[1].c_str()));
167  sources.push_back((float) atof(elems[2].c_str()));
168  sources.push_back((float) atof(elems[3].c_str()));
169  sources.push_back((float) atof(elems[4].c_str()));
170  sources.push_back((float) atof(elems[5].c_str()));
171  sources.push_back((float) atof(elems[6].c_str()));
172  //sources.push_back((float) atof(elems[7].c_str()));
173  }
174  }
175  myfile.close();
176  fillSSBO(sources.data(), sources.size()*sizeof(float),2);
177

179  _computeShader->setUniform("haveCustomSources",(unsigned int)1);
