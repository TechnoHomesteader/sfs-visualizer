35 void ArcBall_t::_mapToSphere(const Point2fT* NewPt, Vector3fT* NewVec) const
36 {
37  Point2fT TempPt;
38  GLfloat length;
39 
40  //Copy paramter into temp point
41  TempPt = *NewPt;
42 
43  //Adjust point coords and scale down to range of [-1 ... 1]
44  TempPt.s.X = (TempPt.s.X * this->AdjustWidth) - 1.0f;
45  TempPt.s.Y = 1.0f - (TempPt.s.Y * this->AdjustHeight);
46 
47  //Compute the square of the length of the vector to the point from the center
48  length = (TempPt.s.X * TempPt.s.X) + (TempPt.s.Y * TempPt.s.Y);
49 
50  //If the point is mapped outside of the sphere... (length > radius squared)
51  if (length > 1.0f)
52  {
53  GLfloat norm;
54 
55  //Compute a normalizing factor (radius / sqrt(length))
56  norm = 1.0f / FuncSqrt(length);
57 
58  //Return the "normalized" vector, a point on the sphere
59  NewVec->s.X = TempPt.s.X * norm;
60  NewVec->s.Y = TempPt.s.Y * norm;
61  NewVec->s.Z = 0.0f;
62  }
63  else //Else it's on the inside
64  {
65  //Return a vector to a point mapped inside the sphere sqrt(radius squared - length)
66  NewVec->s.X = TempPt.s.X;
67  NewVec->s.Y = TempPt.s.Y;
68  NewVec->s.Z = FuncSqrt(1.0f - length);
69  }
70 }
